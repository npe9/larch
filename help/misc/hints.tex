\item \dlink{hints_formalizing}{Formalizing axioms and conjectures}
\item \dlink{hints_ordering}{Orienting formulas into rewrite rules}
\item \dlink{hints_proofs}{Managing proofs}
\item \dlink{hints_speed}{Making proofs go faster}
\end{itemize}
